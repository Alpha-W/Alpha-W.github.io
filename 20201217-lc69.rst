   class Solution {
    public int mySqrt(int x) {
        if(x==0) return 0;
        double s=x/2.0;
        while(s*s-x>=0.0001||s*s-x<=-0.0001)
        s=0.5*(s+x/s);
        return (int)s;
        
    }
   }
