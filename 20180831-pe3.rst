   public class Main {
    public static boolean isprime(long n)
    {
        boolean f=true;
        long sqrtn=1+(long)Math.sqrt(n);
        for(int i=2;i<sqrtn;i++)
            if(n%i==0) f=false;
            return f;
    }
    public static void main(String[] args) {
        long s=600851475;
        long i;
        for(i=800000;i>0;i--)
            if(((1000*(s%i)+143%i)%i==0)&&isprime(i)) break;
            System.out.println(i);
    }
   }//6857
