   import java.util.Stack;
   class Solution {
    public static boolean isValid(String s) {
        int len=s.length();
        if(len<2) return false;
        Stack<Character> st=new Stack<Character>();
        st.push('\0');
        for(int i=0;i<len;i++)
        {
          Character ch =s.charAt(i);
          if((st.peek()=='{'&&ch=='}')||(st.peek()=='['&&ch==']')||(st.peek()=='('&&ch==')')) st.pop();
          else
          {
            st.push(ch);
          }
        }
        st.pop();
       return st.empty();
   }
   }
