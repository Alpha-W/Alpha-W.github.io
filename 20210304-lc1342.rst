<https://leetcode.com/problems/number-of-steps-to-reduce-a-number-to-zero/>`_

.. code:: java

   class Solution {
    public int numberOfSteps (int num) {
        int sum=0;
        if(num==0) return 0;
        while(num!=0)
        {
           if((num&1)==0) sum+=1;
           else sum+=2;
           num>>>=1; 
        }
        return sum-1;
    }
   }
