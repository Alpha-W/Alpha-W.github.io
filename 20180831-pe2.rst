   public class Main {

    public static long fib(int n) {
        long[] a = {1, 1};
        if (n == 1) return a[0];
        else if (n == 2) return a[1];
        else {
            for (int i = n; i > 2; i = i - 2) {
                a[0] = a[0] + a[1];
                a[1] = a[0] + a[1];
            }
            if (n % 2 == 1) return a[0];
            else return a[1];
        }
    }
    public static void main(String[] args) {
        long s=0;
        for(int i=1;fib(i)<4000000;i++) if(fib(i)%2==0) s+=fib(i);
        System.out.println(s);
    }
   }//4613732
