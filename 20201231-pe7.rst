   public class Main {
    public static boolean isprime(long n)
    {
        boolean f=true;
        long sqrtn=1+(long)Math.sqrt(n);
        for(int i=2;i<sqrtn;i++)
            if(n%i==0) f=false;
        return f;
    }
    public static void main(String[] args) {
        long[] p=new long[10001];
        p[0]=2;p[1]=3;
        for (int i = 5,k=2; k<10001 ; i+=2)
            if(isprime(i)) p[k++]=i;
        System.out.println(p[10000]);
    }
   }//104743
