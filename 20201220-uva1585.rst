   import java.util.Scanner;
   public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num=sc.nextInt();
        String s=sc.nextLine();
        int len,sum,k;
        for(int i=0;i<num;i++)
        {
            s=sc.nextLine();
            len=s.length();
            sum=0;k=0;
            for(int j=0;j<len;j++)
            {
             if(s.charAt(j)=='O')
             {
                 k++;
                 sum+=k;
             }
             else k=0;
            }
            System.out.println(sum);
        }
        }
    }
