   class Solution:
    def reverse(self, x):
        """
        :type x: int
        :rtype: int
        """
        if x<0:
            t=-x
            return -(self.reverse(t))
        else:
            s=int((str(x))[::-1])
            if s<=0 or s>=2**31:
                return 0
            else:
                return s
