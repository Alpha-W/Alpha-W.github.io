   class Solution {
    public boolean isPalindrome(int x) {
        if(x<0) return false;
        else if(x==0) return true;
        else
        {
        int n=1+(int)(Math.log10(x));
        int[] a=new int[n];
        for(int i=0;i<n;i++) a[i]=(x/((int)Math.pow(10,i)))%10;
        for(int i=0;i<n;i++) if(a[i]!=a[n-1-i]) return false;
        return true;
        }     
    }
   }
